    <input id="itemInput" type="text" placeholder="اكتب اسم الصنف..." />
    <button onclick="lookupPrice()">عرض السعر</button>
    <div id="result" class="price"></div>
  </div>

  <script>
    const prices = {
      "تفاح": 300,
      "موز": 250,
      "برتقال": 280,
    };

    function lookupPrice() {
      const item = document.getElementById("itemInput").value.trim();
      const resultDiv = document.getElementById("result");

      if (item in prices) {
        resultDiv.textContent = `سعر ${item} هو: ${prices[item]} ريال`;
      } else {
        resultDiv.textContent = "الصنف غير موجود";
      }
    }
  </script>
